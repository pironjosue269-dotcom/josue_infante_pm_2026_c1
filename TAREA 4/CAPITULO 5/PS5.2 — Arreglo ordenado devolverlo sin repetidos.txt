#include <stdio.h>

int main(void){
    int N;

    printf("N: ");
    scanf("%d", &N);

    int V[200], U[200], m=0;

    printf("Ingrese arreglo ORDENADO ascendente:\n");
    for(int i=0;i<N;i++){
        printf("V[%d]: ", i);
        scanf("%d", &V[i]);
    }

    for(int i=0;i<N;i++){
        if(i==0 || V[i]!=V[i-1]){
            U[m++] = V[i];
        }
    }

    printf("Sin repetidos:\n");
    for(int i=0;i<m;i++) printf("%d%s", U[i], (i==m-1)?"\n":" ");

    return 0;
}
