#include <stdio.h>

int main(void){
    unsigned long long F[100];
    F[0]=0; F[1]=1;

    for(int i=2;i<100;i++){
        F[i] = F[i-1] + F[i-2];
    }

    printf("Primeros 100 Fibonacci:\n");
    for(int i=0;i<100;i++){
        printf("%llu%s", F[i], (i%5==4) ? "\n" : "\t");
    }
    if(100%5!=0) printf("\n");
    return 0;
}
