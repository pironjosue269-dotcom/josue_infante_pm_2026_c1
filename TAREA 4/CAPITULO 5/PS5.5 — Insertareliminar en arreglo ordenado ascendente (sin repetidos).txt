#include <stdio.h>

int contiene(int V[], int N, int x){
    for(int i=0;i<N;i++) if(V[i]==x) return 1;
    return 0;
}

int main(void){
    int N, op, ELE;
    int V[200];

    printf("N: ");
    scanf("%d",&N);

    printf("Ingrese arreglo ORDENADO ascendente:\n");
    for(int i=0;i<N;i++){
        printf("V[%d]: ", i);
        scanf("%d",&V[i]);
    }

    printf("1=Insertar  2=Eliminar: ");
    scanf("%d",&op);
    printf("ELE: ");
    scanf("%d",&ELE);

    if(op==1){
        if(contiene(V,N,ELE)){
            printf("No se inserta: ya existe.\n");
        }else{
            int i=N-1;
            while(i>=0 && V[i]>ELE){
                V[i+1]=V[i];
                i--;
            }
            V[i+1]=ELE;
            N++;
            printf("Insertado.\n");
        }
    }else if(op==2){
        int idx=-1;
        for(int i=0;i<N;i++) if(V[i]==ELE){ idx=i; break; }
        if(idx==-1){
            printf("No se elimina: no existe.\n");
        }else{
            for(int i=idx;i<N-1;i++) V[i]=V[i+1];
            N--;
            printf("Eliminado.\n");
        }
    }else{
        printf("Operacion invalida.\n");
        return 0;
    }

    printf("Arreglo final:\n");
    for(int i=0;i<N;i++) printf("%d%s", V[i], (i==N-1)?"\n":" ");

    return 0;
}
