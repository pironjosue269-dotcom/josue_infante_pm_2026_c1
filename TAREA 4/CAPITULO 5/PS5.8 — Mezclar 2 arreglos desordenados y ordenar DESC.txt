#include <stdio.h>

void ordenar_desc(int V[], int n){
    for(int i=0;i<n-1;i++)
        for(int j=0;j<n-1-i;j++)
            if(V[j] < V[j+1]){
                int t=V[j]; V[j]=V[j+1]; V[j+1]=t;
            }
}

int main(void){
    int N,M, C[300];

    printf("N: "); scanf("%d",&N);
    printf("M: "); scanf("%d",&M);

    printf("VEC1:\n");
    for(int i=0;i<N;i++){ printf("A[%d]: ", i); scanf("%d",&C[i]); }

    printf("VEC2:\n");
    for(int i=0;i<M;i++){ printf("B[%d]: ", i); scanf("%d",&C[N+i]); }

    ordenar_desc(C, N+M);

    printf("Mezclado DESC:\n");
    for(int i=0;i<N+M;i++) printf("%d%s", C[i], (i==N+M-1)?"\n":" ");

    return 0;
}
