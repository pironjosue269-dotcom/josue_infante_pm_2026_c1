#include <stdio.h>

int es_primo(int n){
    if(n < 2) return 0;
    if(n == 2) return 1;
    if(n % 2 == 0) return 0;
    for(int i=3; i*i<=n; i+=2){
        if(n % i == 0) return 0;
    }
    return 1;
}

int main(void){
    int NUM;
    printf("NUM: ");
    scanf("%d", &NUM);

    if(NUM <= 0){
        printf("Debe ser positivo\n");
        return 0;
    }

    printf("%d %s primo\n", NUM, es_primo(NUM) ? "ES" : "NO ES");
    return 0;
}
