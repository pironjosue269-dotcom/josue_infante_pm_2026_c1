#include <stdio.h>

int main(void){
    int N;
    printf("Numero de 4 digitos: ");
    scanf("%d", &N);

    if(N < 1000 || N > 9999){
        printf("Debe ser de 4 digitos\n");
        return 0;
    }

    int d1 = N/1000;
    int d2 = (N/100)%10;
    int d3 = (N/10)%10;
    int d4 = N%10;

    printf("Invertido: %d%d%d%d\n", d4,d3,d2,d1);
    return 0;
}
